import sys
from collecons import defaultdict
# Diconary to store intermediate values
intermediate = defaultdict(list)
for line in sys.stdin:
line = line.strip()
key, value = line.split("\t")
i, j = map(int, key.split())
parts = value.split()
intermediate[(i, j)].append((parts[0], int(parts[1]), int(parts[2])))
# Compute nal matrix mulplicaon
for (i, j), values in intermediate.items():
a_values = {k: v for m, k, v in values if m == "A"}
b_values = {k: v for m, k, v in values if m == "B"}
result = sum(a_values[k] * b_values[k] for k in a_values if k in b_values)
if result:
print(f"{i} {j} {result}")

from mrjob.job import MRJob
class WeatherAnalysis(MRJob):
def mapper(self, _, line):
try:
date, temp, humidity, precipitaon = line.split(',')
temp = oat(temp)
humidity = oat(humidity)
precipitaon = oat(precipitaon)
yield date, (temp, humidity, precipitaon)
except ValueError:
pass # Skip lines with errors
def reducer(self, date, values):
total_temp, total_humidity, total_precipitaon = 0, 0, 0
count = 0
for temp, humidity, precipitaon in values:
total_temp += temp
total_humidity += humidity
total_precipitaon += precipitaon
count += 1
avg_temp = total_temp / count
avg_humidity = total_humidity / count
avg_precipitaon = total_precipitaon / count
# Determine weather condion
if avg_temp > 30:
condion = "Hot Day"
elif avg_temp < 10:
condion = "Cold Day"
elif avg_precipitaon > 5:
condion = "Rainy Day"
elif avg_humidity > 80:
condion = "Humid Day"
else:
condion = "Pleasant Day"
yield date, f"{condion} - Avg Temp: {avg_temp:.1f}C, Humidity:
{avg_humidity:.1f}%, Precipitaon: {avg_precipitaon:.1f}mm"
if __name__ == "__main__":
WeatherAnalysis.run()

from mrjob.job import MRJob
import re
WORD_RE = re.compile(r"\b\w+\b")
class WordCountMR(MRJob):
def mapper(self, _, line):
"""Emit each word with a count of 1"""
for word in WORD_RE.ndall(line):
yield word.lower(), 1
def reducer(self, word, counts):
"""Sum up counts for each word"""
yield word, sum(counts)
if __name__ == "__main__":
WordCountMR.run()

word count program in spark
To create a le

from pyspark.sql import SparkSession
from pyspark.sql.funcons import explode, split, col
# Inialize Spark session
spark = SparkSession.builder.appName("WordCount").getOrCreate()
# Read input text le
text_rdd = spark.read.text("input.txt").rdd
# Perform word count using RDD transformaons
word_counts = (
text_rdd.atMap(lambda line: line[0].split()) # Split lines into words
.map(lambda word: (word.lower(), 1)) # Map each word to (word, 1)
.reduceByKey(lambda a, b: a + b) # Reduce by key (word)
)
# Collect and print results
for word, count in word_counts.collect():
print(f"{word}: {count}")
# Stop Spark session
spark.stop()
